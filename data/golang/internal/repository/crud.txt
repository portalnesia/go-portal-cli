package repository

import (
	iface "APP_NAME/internal/interface"

	"github.com/uptrace/bun"
)

type CrudRepository[Model any] interface {
	Create(ctx iface.Context, entity *Model) error
	GetByPK(ctx iface.Context, entity *Model, opt ...WithFn) (*Model, error)

	UpdateByPK(ctx iface.Context, entity *Model, opt ...WithFn) error
	DeleteByPK(ctx iface.Context, entity *Model) error
	UpsertByPK(ctx iface.Context, entity *Model, opt ...WithFn) error
	FirstOrUpsertByPK(ctx iface.Context, entity *Model, opt ...WithFn) (*Model, error)
}

type crudRepository[Model any] struct {
	*base
}

func (r *crudRepository[Model]) Create(ctx iface.Context, entity *Model) error {
	db := r.getDatabase(ctx)
	_, err := db.NewInsert().
		Model(entity).
		Exec(ctx.DBContext())
	return err
}

func (r *crudRepository[Model]) GetByPK(ctx iface.Context, entity *Model, withopt ...WithFn) (*Model, error) {
	opt := withOptions(withopt...)
	return dbGet(r.base, ctx, entity, func(db *bun.SelectQuery) *bun.SelectQuery {
		return db.WherePK()
	}, opt)
}

func (r *crudRepository[Model]) UpdateByPK(ctx iface.Context, entity *Model, opt ...WithFn) error {
	op := withOptions(opt...)
	return dbUpdate(r.base, ctx, entity, func(db *bun.UpdateQuery) *bun.UpdateQuery {
		return db.WherePK()
	}, op)
}

func (r *crudRepository[Model]) DeleteByPK(ctx iface.Context, entity *Model) error {
	return dbDelete(r.base, ctx, entity, func(db *bun.DeleteQuery) *bun.DeleteQuery {
		return db.WherePK()
	}, DBOpt{})
}

func (r *crudRepository[Model]) UpsertByPK(ctx iface.Context, entity *Model, opt ...WithFn) error {
	op := withOptions(opt...)
	return dbUpsert(r.base, ctx, entity, func(db *bun.SelectQuery) *bun.SelectQuery {
		return db.WherePK()
	}, op)
}

func (r *crudRepository[Model]) FirstOrUpsertByPK(ctx iface.Context, entity *Model, opt ...WithFn) (*Model, error) {
	op := withOptions(opt...)
	data, err := dbFirstOrUpsert[Model](r.base, ctx, entity, func(db *bun.SelectQuery) *bun.SelectQuery {
		return db.WherePK()
	}, op)

	return data, err
}
