package repository

import (
	"context"
	iface "APP_NAME/internal/interface"

	"github.com/uptrace/bun"
)

type BaseRepository interface {
	StartTransaction(c iface.Context, fc func() error) (err error)
}

type base struct {
	db      bun.IDB
  log     iface.Logger
  env     iface.Env
}

type dbValue struct{}

func (r base) StartTransaction(c iface.Context, fc func() error) (err error) {
	err = r.db.RunInTx(c.Context(), nil, func(ctx context.Context, tx bun.Tx) error {
		// save transaction to context
		ctx = context.WithValue(ctx, dbValue{}, tx)
		c.DBContext(ctx)
		errTx := fc()
		if errTx != nil {
			return errTx
		}
		return nil
	})
	c.DBContext(nil)
	return err
}

func (r base) DB() bun.IDB {
	return r.db
}

func (r base) getDatabase(ctx iface.Context) bun.IDB {
	var getDB = func() bun.IDB {
		dbAny := ctx.DBContext().Value(dbValue{})
		if dbAny != nil {
			dbbun, ok := dbAny.(bun.IDB)
			if ok {
				return dbbun
			}
		}

		return r.db
	}

	db := getDB()

	return db
}

func (r base) GetRedisKeyf(key string, args ...interface{}) string {
	return r.redis.GetRedisKeyf("db_cached:"+key, args...)
}
