package repository

import (
	iface "APP_NAME/internal/interface"

	"github.com/uptrace/bun"
)

type Registry struct {
	Base BaseRepository
}

type NewRepositoryConfig struct {
	DB    bun.IDB
	Env   iface.Env
	Log   iface.Logger
}

func NewRepository(conf NewRepositoryConfig) Registry {
	bs := base{conf.DB, conf.Log, conf.Env}
	return Registry{
		Base: bs,
	}
}
